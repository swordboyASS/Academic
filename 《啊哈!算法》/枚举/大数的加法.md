
#### :star: 要点:用字符数组减小空间,存储进位数.
```c
#include<stdio.h>
#include<string.h>
#define max 102
int main()
{
	char a[max],b[max],c[max];  //a,b两个数都不超过100位。
	memset(a,0,max);
	memset(b,0,max);
	memset(c,0,max);	//将a,b,c全部初始化位0； 
	
	int r=0,i,j=0,num,lena,lenb;
	scanf("%s",a);
	scanf("%s",b);
	lena = strlen(a);
	lenb = strlen(b);
	
	   do{
    	c[j]=(a[lena-1]+b[lenb-1]-96+r)%10;
        r=(a[lena-1]+b[lenb-1]-96+r)/10;
		j++;
        lena--;
        lenb--;
        
	}while(lena>0&&lenb>0);
	while(lena>0)
	{
		c[j]=(a[lena-1]+r-48)%10;
		r=(a[lena-1]+r-48)/10;
		lena--;
		j++;	
	}
	while(lenb>0)
	{
		c[j]=(b[lenb-1]+r-48)%10;
		r=(b[lenb-1]+r-48)/10;
		lenb--;
		j++;	
	}
	if(r>0)
	{
		c[j]=r;
		for(i=j;i>=0;i--)
		{
			printf("%d",c[i]);
		}
	}
	else
	{
		for(i=j-1;i>=0;i--)
		{
			printf("%d",c[i]);
		}
	}
	return 0;
}
```
