
#### 数组模拟栈
```c
#include<stdio.h>
int main()
{
	int a[101]={6,3,1,7,5,8,9,2,4},head,tail;   // tail 指向队尾的后一个 
	head = 0;
	int b=0;
	tail = 9;
	while(head<tail){		//当head=tail时，列表为空。 
		
		//打印队首并将队首移出
		 b= a[head];
		printf("%d ",b);
		head++;
		
		//将新队首添加到队尾,然后队首往后移 
		a[tail] = a[head];
		tail++; //因为有新元素添加，所以队尾往后移。 
		head++;
		 
	} 	
		
	return 0;	
}

```
