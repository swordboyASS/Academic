	int t;

	h[x]=h[y];
	h[y]=t;

#include <iostream>
using namespace std;
int h[101]; //用于存放堆的数组
int n;//用于存储堆的节点数
//交换函数，交换堆中2个元素的值
void swap_(int x,int y)
{
	int t;
	t=h[x];
	h[x]=h[y];
	h[y]=t;
 } 
 
 //向下调整函数
 void siftdown(int i) //传入一个需要向下调整的节点编号i,这里传入1,即从堆顶开始
 {
 	int t,flag=0; //flag用于判断是否需要继续向下调整。
	 while(i*2<=n && flag==0)
	 {
	 	//首先判断与左儿子的关系,并用t记录更大的数的编号
		if(h[i] < h[i*2])
		 	t=i*2;
		else
			t=i;
		//如果有右儿子，再对右儿子进行讨论，
		if(i*2+1<=n)
		{
			//如果右儿子的值更大，更新较小的节点编号 
			if(h[t] < h[2*i+1])
			t=i*2+1; 
		}
		
		if(t!=i)
		{
			swap_(t,i);
			i=t;	//继续向下调整 
		}
		else
			flag=1;	//说明已经调整好了。不再需要继续调整 
	  } 
  } 
//创建堆的函数，建立出来时大顶堆 
void create()
{
	int i;
	//从最后一个非叶节点到第1个节点进行向下调整
	for(i=n/2;i>=1;i--)
	{
		siftdown(i);
	 } 
 }
