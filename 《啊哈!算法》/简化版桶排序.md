///summary
///低级的桶排序 
///summary
#include<stdio.h>

int main(void)
{
	int a[10]={0};   //初始化数组全为0 
	int i=0,
	j=1;
	int n=10;
	int times; 		//用于记录下标的数。 
	while(n>0){
		scanf("%d",&times);
		a[times]++;
		n--;
	}
	for(i=0;i<10;i++){   //必须像这样写完整，不然会有错误。 
		for(j=1;j<=a[i];j++){
			printf("%d ",i);
		}
	}
	return 0;
}
