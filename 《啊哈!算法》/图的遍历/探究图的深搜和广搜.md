
- [x] 无向图的邻接矩阵存储无向图
#### :star: 1. 深搜
```c++
#include <iostream>
using namespace std;

int book[101],sum=0,n,e[101][101];
void dfs(int cur)
{
	int i;
	cout<<cur<<" "; //输出访问过的点
	sum++;
	if(sum==n)  return;
	
	for(i=1;i<=n;i++)
	{
		//判断是否可达，和是否访问过 
		if(e[cur][i]==1 && book[i]==0)
		{
			book[i]=1;
			dfs(i); //再从顶点i出发继续遍历 
		}
	}
	return ; 
}

int main()
{
	int i,j,m,a,b;
	cin>>n>>m;  //n表示顶点数，m表示m个顶点之间有关系 
	//初始化二维矩阵：对角线为0,其余先初始化为99999999； 
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			if(i==j) e[i][j]=0;
			else e[i][j]=99999999;
	
	for(i=1;i<=m;i++)
	{
		cin>>a>>b;
		e[a][b]=1;
		e[b][a]=1; //无向图 
	}
	book[1]=1;
	dfs(1);
	return 0;
}
```
