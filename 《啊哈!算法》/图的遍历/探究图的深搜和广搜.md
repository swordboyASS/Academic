#### :star: 1. 深搜
```c++
#include <iostream>
using namespace std;

int book[101],sum=0,n,e[101][101];
void dfs(int cur)
{
	int i;
	cout<<cur<<" "; //输出访问过的点
	sum++;
	if(sum==n)  return;
	
	for(i=1;i<=n;i++)
	{
		//判断是否可达，和是否访问过 
		if(e[cur][i]==1 && book[i]==0)
		{
			book[i]=1;
			dfs(i); //再从顶点i出发继续遍历 
		}
	}
	return ; 
}

int main()
{
	int i,j,m,a,b;
	cin>>n>>m;  //n表示顶点数，m表示m个顶点之间有关系 
	//初始化二维矩阵：对角线为0,其余先初始化为99999999； 
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			if(i==j) e[i][j]=0;
			else e[i][j]=99999999;
	
	for(i=1;i<=m;i++)
	{
		cin>>a>>b;
		e[a][b]=1;
		e[b][a]=1; //无向图 
	}
	book[1]=1;
	dfs(1);
	return 0;
}
```

#### :star: 2.广搜遍历无向图
```c++
#include <iostream>
using namespace std;
int main()
{
	int i,j,n,m,a,b,cur,book[101]={0},e[101][101];
	int que[10001],head,tail;
	cin>>n>>m;
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			if(j==i) e[i][j]=0;
			else e[i][j]=99999999;
	//读入边
	for(i=1;i<=m;i++)
	{
		cin>>a>>b;
		e[a][b]=1;
		e[b][a]=1;	
	}	
	
	//队列初始化
	head=1;
	tail=1;
	
	//从一号顶点出发，
	que[tail]=1;
	tail++;
	book[1]=1;	//标记1号顶点已访问 
	
	//当队列不为空时循环
	while(head<tail)
	{
		cur=que[head]; //当前正在访问的顶点编号
		for(i=1;i<=n;i++)
		{
				//判断从顶点cur到顶点i是否有边，并判断i是否被访问过
				if(e[cur][i]==1 && book[i]==0)
				{
					que[tail]=i;
					tail++;
					book[i]=1; //标记i已经访问过了	
				} 
				//如果tail大于n表示所有顶点都已经访问过了
				if(tail>n) break;  
		}
		head++;
	 } 
	 for(i=1;i<tail;i++)
	 {
	 	cout<<que[i];
	 }
	return 0;
}
```
