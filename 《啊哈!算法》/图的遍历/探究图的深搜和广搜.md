#### :star: 1. 深搜，使用邻接矩阵存储
```c++
#include <iostream>
using namespace std;

int book[101],sum=0,n,e[101][101];
void dfs(int cur)
{
	int i;
	cout<<cur<<" "; //输出访问过的点
	sum++;
	if(sum==n)  return;
	
	for(i=1;i<=n;i++)
	{
		//判断是否可达，和是否访问过 
		if(e[cur][i]==1 && book[i]==0)
		{
			book[i]=1;
			dfs(i); //再从顶点i出发继续遍历 
		}
	}
	return ; 
}

int main()
{
	int i,j,m,a,b;
	cin>>n>>m;  //n表示顶点数，m表示m个顶点之间有关系 
	//初始化二维矩阵：对角线为0,其余先初始化为99999999； 
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			if(i==j) e[i][j]=0;
			else e[i][j]=99999999;
	
	for(i=1;i<=m;i++)
	{
		cin>>a>>b;
		e[a][b]=1;
		e[b][a]=1; //无向图 
	}
	book[1]=1;
	dfs(1);
	return 0;
}
```

#### :star: 2.广搜遍历无向图,使用邻接矩阵存储
```c++
#include <iostream>
using namespace std;
int main()
{
	int i,j,n,m,a,b,cur,book[101]={0},e[101][101];
	int que[10001],head,tail;
	cin>>n>>m;
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			if(j==i) e[i][j]=0;
			else e[i][j]=99999999;
	//读入边
	for(i=1;i<=m;i++)
	{
		cin>>a>>b;
		e[a][b]=1;
		e[b][a]=1;	
	}	
	
	//队列初始化
	head=1;
	tail=1;
	
	//从一号顶点出发，
	que[tail]=1;
	tail++;
	book[1]=1;	//标记1号顶点已访问 
	
	//当队列不为空时循环
	while(head<tail)
	{
		cur=que[head]; //当前正在访问的顶点编号
		for(i=1;i<=n;i++)
		{
				//判断从顶点cur到顶点i是否有边，并判断i是否被访问过
				if(e[cur][i]==1 && book[i]==0)
				{
					que[tail]=i;
					tail++;
					book[i]=1; //标记i已经访问过了	
				} 
				//如果tail大于n表示所有顶点都已经访问过了
				if(tail>n) break;  
		}
		head++;
	 } 
	 for(i=1;i<tail;i++)
	 {
	 	cout<<que[i];
	 }
	return 0;
}
```

#### :star: 3.城市路径问题--有向图的最短路径,深搜,使用邻接矩阵存储
```c++
#include <iostream>
using namespace std;

int min_=9999999999,book[101],n,e[101][101]; 
void dfs(int cur,int dis_)
{
	int j;
	//如果当前走过的路径已经大于之前的最短路径，则没有继续往下尝试的必要，直接返回 
	if(dis_>min_) return;
	if(cur==n) //判断是否到达了目标城市
	{
		if(dis_<min_) min_=dis_;
		return ;
	} 
	for(j=1;j<=n;j++)
	{
		//判断当前城市cur到城市j是否有通路，并判断城市j是否已在走过的路中 
		if(e[cur][j]!=99999999 && book[j]==0)
		{
			book[j]=1; //标记城市已在路径中 
			dfs(j,dis_+e[cur][j]); //从城市j出发，继续寻找目标城市 
			book[j]=0;	//之前一步探索完毕之后，取消对城市j的标记 
		}
	}
	return ;
	 
}

int main()
{
	int i,j,m,a,b,c;
	cin>>n>>m;
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			if(i==j) e[i][j]=0;
			else e[i][j]=99999999;
	for(i=1;i<=m;i++)
	{
		cin>>a>>b>>c; 
		e[a][b]=c; 
	}

	//从1号城市出发
	book[1]=1;
	dfs(1,0);
	cout<<min_; 
	
	return 0;
 } 
```

#### :star: 4. 最少转飞机次数-图的广度优先遍历,邻接表存储 page:143
```c++
#include <iostream>
using namespace std;

struct note
{
	int x; //城市编号
	int s; //转机次数	
};

int main()
{
	struct note que[2051];
	int e[51][51]={0},book[51]={0};
	int head,tail;
	int i,j,n,m,a,b,cur,start,end,flag=0;
	
	cin>>n>>m>>start>>end;
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			if(i==j) e[i][j]=0;
			else e[i][j]=99999999;
	for(i=1;i<=m;i++)
	{
		cin>>a>>b; 
		e[a][b]=1;
		e[b][a]=1; 
	}

	//队列初始化
	head=1;
	tail=1;
	
	//从start城市出发，将start号城市入队列
	que[tail].x=start;
	que[tail].s=0;
	tail++;
	book[start]=1; //标记start号城市已在队列中
	
	//当队列不为空的时候循环
	while(head<tail)
	{
		cur = que[head].x; //当前队首城市的编号
		for(j=1;j<=n;j++) //从1到n依次尝试 
		{
			//从城市cur到城市j是否有航班并且判断城市j是否已在队列中
			if(e[cur][j]!=99999999 && book[j]==0)
			{
				//如果从城市cur到城市j有航班并且城市j不在队列中，将j入队列
				que[tail].x=j;
				que[tail].s=que[head].s+1;
				tail++;
				//标记城市j已经在队列中
				book[j]=1; 
			 } 
			 
			 //如果到达目标城市，停止扩展，任务结束，退出循环
			 if(que[tail-1].x==end)
			 {
			 	flag=1;
			 	break;
			  } 
		 } 
		 if(flag==1)
		 	break;
		head++;
	 } 
	cout<<que[tail-1].s; //打印最后的转机次数
	
	/*验证数据
	5  7 1 5 
	1 2  
	1 3
	2 3
	2 4
	3 4
	3 5
	4 5 
	*/ 
	return 0;
 }
```
