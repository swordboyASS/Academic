#### :star: 3. 3个数排序
```c
int m,n,p,t=0;
scanf("%d%d%d",&m,&n,&p);
if(m>n)
{
	t = m; m = n; n = t;  //此时m<=n 
}
if(m>p)
{
	t = m; m = p; p = t; //此时m<=p,m<=n 
}
if(n>p)
{
	t = n; n = p; p = t; 
}
printf("%d %d %d",m,n,p);
```
